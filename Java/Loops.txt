C-I Can Use Both Hands!

import java.util.Scanner;
public class Main {
	public static void main(String[]args) {
		Scanner sc=new Scanner(System.in);
		int T;
		long L,R,M,pages,lines;
		T=sc.nextInt();
		while(T-->0) {
			 L=sc.nextLong();
			 R=sc.nextLong();
			 M=sc.nextLong();
			 pages=(M+L-1)/L;
			 lines=M/R;
			System.out.println(pages+lines);
		}	
	}
}

D-Chef and Stocks

import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T= sc.nextInt();
        while (T-- > 0) {
            int days = sc.nextInt();
            int totalProfit = 0;
            int smallestProfit = Integer.MAX_VALUE;
            for (int i = 0; i < days; i++) {
                int profit = sc.nextInt();
                totalProfit += profit;
                if (profit < smallestProfit) {
                    smallestProfit = profit;
                }
            }
            int maxProfit = totalProfit - smallestProfit;
            System.out.println(maxProfit);
        }
    }
}
